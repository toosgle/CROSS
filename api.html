定义组件

CROSS.define( name, object, superClass )

参数

  • name { String } 组件名（web组件以'Web'开头，native组件以'Native'开头，组件名首字母大写）
  • object { Object } 组件方法
  • object { Object } 组件基类
// 定义web组件
CROSS.define( 'WebSlider', {
    _init: function(){
        var me = this;

        // 进行创建DOM等操作
    },
    toNext: function(){
        this._instance.toNext();
    }
}, CROSS.WebBase);

// 定义native组件
CROSS.define( 'NativeSlider', {
    toNext: function(){
        this._exec( 'toNext', function(){}, function(){}, [] );
    }
}, CROSS.NativeBase);

调用组件

CROSS.use( name, callback )

参数

  • name { String | Array } 组件名
  • callback { Function } 组件调用成功后的回调函数
//使用CROSS.use声明对一个组件的调用
CROSS.use('slider', function(slider){
    //使用slider.create来实例化Slider组件
    var sliderInstance = slider.create(config);
    //调用Slider组件的公共方法
    sliderInstance.toNext();
});
//使用CROSS.use声明对多个组件的调用
CROSS.use(['slider', 'gallery'], function(slider, gallery){
    //使用slider.create来实例化Slider组件
    var sliderInstance = slider.create(config);
    //调用Slider组件的公共方法
    sliderInstance.toNext();

    //使用gallery.create来实例化Gallery组件
    var galleryInstance = gallery.create(config);
    //调用Gallery组件的公共方法
    galleryInstance.toNext();
});

公共方法

订阅事件

instance.on( name, callback[, context] )

参数

  • name { String } 事件名
  • callback { Function } 事件处理器
  • context { Object } 事件处理器的上下文

返回值

  • { Object } 实例本身

订阅事件（只执行一次）

instance.one( name, callback[, context] )

参数

  • name { String } 事件名
  • callback { Function } 事件处理器
  • context { Object } 事件处理器的上下文

返回值

  • { Object } 实例本身

解除事件监听

instance.off( name, callback[, context] )

参数

  • name { String } 事件名
  • callback { Function } 事件处理器
  • context { Object } 事件处理器的上下文

返回值

  • { Object } 实例本身

触发事件

instance.trigger( name[, ...] )

参数

  • name { String } 事件名
  • callback { Function } 事件处理器
  • context { Object } 事件处理器的上下文

返回值

  • { Object } 实例本身

判断组件是否是Native组件

instance.isNative()

参数（无）

返回值

  • { Boolean } 组件是否是Native组件
